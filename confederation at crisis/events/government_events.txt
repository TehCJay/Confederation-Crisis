character_event = {
	id = government.100
	desc = "EVTDESCGOVERNMENT100"
	picture = GFX_evt_government
	
	only_rulers = yes
	
	trigger = {
		independent = no
		is_landed = yes
		is_government_potential = military_monarchy
		OR = {
			government = military_republic
			government = military_directorate
		}
		
		# Some legitimacy law applies to you
		has_crown_law_title = yes
		crownlaw_title = {
			has_law = military_vassal_legitimacy_0
			has_law = dote_autonomy_0
			OR = {
				has_law = centralization_4
				AND = {
					has_law = centralization_3
					holder_scope = { any_vassal = { character = ROOT } }
				}
				AND = {
					has_law = centralization_2
					holder_scope = { any_vassal = { character = ROOT } }
					ROOT = { NOT = { higher_tier_than = duke } }
				}
				AND = {
					holder_scope = {
						tier = emperor
						any_vassal = { character = ROOT }
					}
					ROOT = { NOT = { higher_tier_than = duke } }
				}
				AND = {
					holder_scope = { any_vassal = { character = ROOT } }
					ROOT = { NOT = { higher_tier_than = count } }
				}
			}
		}
	}
	option = {
		name = "EVTOPTCONFIRM"
		set_government_type = military_monarchy
		any_demesne_title = {
			limit = {
				NOT = { has_law = succ_primogeniture }
				NOT = { has_law = succ_ultimogeniture }
				NOT = { has_law = succ_seniority }
				NOT = { has_law = succ_gavelkind }
				NOT = { has_law = succ_tanistry }
				NOT = { has_law = succ_feudal_elective }
			}
			add_law = succ_gavelkind
		}
		any_demesne_title = {
			limit = { NOT = { has_law = true_cognatic_succession } }
			add_law = true_cognatic_succession
		}
	}
}

# You are military, but you own no castles

	id = government.1000
	desc = "EVTDESCGOVERNMENT1000"
	picture = GFX_evt_government
	
	only_rulers = yes
	
	trigger = {
		OR = {
			government = military_monarchy
			government = military_republic

				holding_type = castle

# Deconverting space pirates should adopt an appropriate government
character_event = {
	id = government.3000
	desc = "EVTDESCGOVERNMENT3000"
	hide_window = yes
	is_triggered_only = yes
	
	trigger = {
		is_ruler = yes
		OR = {
			is_patrician = no
			is_merchant_republic = yes
		}
		OR = {
			governemnt = pirate_kingdom
			government = pirate_republic
		}
	}
	
	immediate = {
		if = {
			limit = { governemnt = pirate_republic }
			set_government_type = merchant_republic_government
		}
		if = {
			limit = {
				government = pirate_kingdom
				capital_holding = { holding_type = castle }
			}
			set_government_type = military_monarchy
			any_demesne_title = { add_law = succ_gavelkind }
		}
		if = {
			limit = {
				government = pirate_kingdom
				capital_holding = { holding_type = city }
			}
			set_government_type = corporate_monarchy
			any_demesne_title = { add_law = succ_gavelkind }
		}
		if = {
			limit = {
				government = pirate_kingdom
				capital_holding = { holding_type = temple }
			}
			set_government_type = administrative_monarchy
			any_demesne_title = { add_law = succ_gavelkind }
		}
	}
}
