	desc = "EVTDESCFLAVOR100"

				factor = 2

			}

				factor = 0
				owner = { religion = xenoterran }
			}
		}

		# Let them stay - lose piety and prestige, civil unrest decreases

		owner = {
			prestige = -50
			piety = -50
		}
		province_event = { id = laws.201 tooltip = "EVTTOOLTIPLAWS201" }
		ai_chance = {
			factor = 50
			modifier = {
				factor = 0
				owner = {
					OR = {
						religion = pilgrim
						religion = orthodox
					}
				}
			}
		}
	}
}

# A nomadic fleet wants to pass through your space
