# Try to convert friend

	id = conversion.5000
	desc = "EVTDESCCONVERSION5000"
	picture = GFX_evt_quarrel
	
	min_age = 16
	only_capable = yes
	prisoner = no
	
	trigger = {
		any_friend = {
			at_location = ROOT
			NOT = { religion = ROOT }
			age = 16
			NOT = { trait = incapable }
			prisoner = no
			NOT = {
				any_war = {
					attacker = { character = ROOT }
					OR = {
						using_cb = democratic_rebellion
						using_cb = autocratic_rebellion
						using_cb = theocratic_rebellion
					}
				}
			}
		}
		NOT = { religion = space_pirate }

		months = 1200
	}
	
	option = {
		name = "EVTOPTACONVERSION5000"
		random_friend = {
			limit = {
				at_location = ROOT
				NOT = { religion = ROOT }
				age = 16
				NOT = { trait = incapable }
				prisoner = no
				NOT = {
					any_war = {
						attacker = { character = ROOT }
						OR = {
							using_cb = democratic_rebellion
							using_cb = autocratic_rebellion
							using_cb = theocratic_rebellion
						}
					}
				}
			}
			character_event = { id = conversion.5001 tooltip = "EVTTOOLTIPCONVERSION5001" }
		}
		piety = 100
		ai_chance = {
			factor = 50
			modifier = {
				factor = 2
				trait = radical
			}
			modifier = {
				factor = 0.5
				trait = pragmatic
			}
			modifier = {
				factor = 2
				trait = extraverted
			}
			modifier = {
				factor = 0.5
				trait = introverted
			}
			modifier = {
				factor = 2
				trait = cold
			}
			modifier = {
				factor = 0.5
				trait = friendly
			}
		}
	}
	
	option = {
		name = "EVTOPTBCONVERSION5000"
		ai_chance = { factor = 50 }
	}
}
