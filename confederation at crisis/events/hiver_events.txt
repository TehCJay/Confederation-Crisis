# Installing a hiver implant (voluntary)
character_event = {
	id = hiver.0
	desc = "EVTDESCHIVER0"
	picture = GFX_evt_cybernetics
	is_triggered_only = yes
	
	trigger = { religion = hiver }
	
	option = {
		name = "EVTOPTAHIVER0"
		add_trait = hiver_implant
		piety = 100
		if = {
			limit = { trait = neurotic }
			random = {
				chance = 20
				add_trait = lunatic
			}
		}
	}
}

# Installing a hiver implant (by force)
character_event = {
	id = hiver.1
	desc = "EVTDESCHIVER1"
	picture = GFX_evt_cybernetics
	is_triggered_only = yes
	
	option = {
		name = "EVTOPTAHIVER1"
		add_trait = hiver_implant
		if = {
			limit = {
				NOT = { religion = hiver }
				NOT = { trait = radical }
			}
			religion = hiver
		}
		if = {
			limit = {
				NOT = { religion = hiver }
				trait = radical
				NOT = { trait = lunatic }
			}
			religion = hiver
			remove_trait = radical
			add_trait = lunatic
		}
		if = {
			limit = { trait = neurotic }
			add_trait = lunatic
		}
		if = {
			limit = { trait = craven }
			add_trait = lunatic
		}
		if = {
			limit = {
				prisoner = yes
				is_landed = no
			}
			host = {
				piety = 100
				ROOT = { prisoner = no }
				reverse_banish = ROOT
			}
		}
		if = {
			limit = {
				prisoner = yes
				is_landed = yes
				OR = {
					AND = {
						tier = baron
						host = { higher_tier_than = baron }
					}
					AND = {
						tier = count
						host = { higher_tier_than = count }
					}
					AND = {
						tier = duke
						host = { higher_tier_than = duke }
					}
					AND = {
						tier = king
						host = { tier = emperor }
					}
				}
			}
			host = {
				piety = 100
				ROOT = {
					prisoner = no
					set_defacto_liege = PREV
				}
			}
		}
		if = {
			limit = {
				prisoner = yes
				is_landed = yes
				NOT = {
					AND = {
						tier = baron
						host = { higher_tier_than = baron }
					}
				}
				NOT = {
					AND = {
						tier = count
						host = { higher_tier_than = count }
					}
				}
				NOT = {
					AND = {
						tier = duke
						host = { higher_tier_than = duke }
					}
				}
				NOT = {
					AND = {
						tier = king
						host = { tier = emperor }
					}
				}
			}
			host = { piety = 100 }
			prisoner = no
		}
	}
}
