			trait = gregarious
		}

			factor = 0.5
			trait = shy
		}

		name = "EVTOPTA76000"

			limit = {
				NOT = {
					trait = lustful
				}
			}

						id = 38272 #Notify Lustful
					}

		}
	}

		name = "EVTOPTB76000"
		if = {

}
