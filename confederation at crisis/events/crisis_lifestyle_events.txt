# Gardener
character_event = {
	id = lifestyle.10
	desc = "EVTDESCLIFESTYLE10"
	picture = GFX_evt_garden
	is_triggered_only = yes
	
	trigger = {
		age = 12
		NOT = { trait = incapable }
		prisoner = no
		NOT = { lifestyle_traits = 1 }
		
		# Must be somewhere where gardening is a plausible hobby
		OR = {
			AND = {
				is_ruler = yes
				capital_holding = { NOT = { is_pressurized_trigger = yes } }
			}
			AND = {
				is_ruler = no
				liege = { capital_holding = { NOT = { is_pressurized_trigger = yes } } }
			}
		}
		in_command = no
	}
	
	weight_multiplier = {
		factor = 1
		modifier = {
			factor = 10
			has_ambition = obj_adopt_a_lifestyle
		}
		modifier = {
			factor = 2
			trait = friendly
		}
		modifier = {
			factor = 0.5
			trait = cold
		}
		modifier = {
			factor = 2
			has_education_stewardship_trigger = yes
		}
	}
	
	option = {
		name = "EVTOPTALIFESTYLE10"
		add_trait = gardener
	}
	option = {
		name = "EVTOPTBLIFESTYLE10"
		ai_chance = { factor = 0 }
	}
}

	id = lifestyle.1403
	desc = "EVTDESCLIFESTYLE1403"
	picture = GFX_evt_falconer
	is_triggered_only = yes
	only_rulers = yes
	
	trigger = {
		is_ruler = yes # Only rulers can afford to breed animals in captivity
		any_artifact = {
			artifact_type = peregrine_falcon
			has_artifact_flag = male
			artifact_age = 3
			NOT = { has_artifact_flag = mated }
		}
		any_artifact = {
			artifact_type = peregrine_falcon
			has_artifact_flag = female
			artifact_age = 3
			NOT = { has_artifact_flag = mated }
		}
	}
	
	immediate = {
		random_artifact = {
			limit = {
				artifact_type = peregrine_falcon
				has_artifact_flag = male
				artifact_age = 3
				NOT = { has_artifact_flag = mated }
			}
			save_event_target_as = male_mate
		}
		random_artifact = {
			limit = {
				artifact_type = peregrine_falcon
				has_artifact_flag = female
				artifact_age = 3
				NOT = { has_artifact_flag = mated }
			}
			save_event_target_as = female_mate
		}
	}
	
	option = {
		# Allow the mating
		name = "EVTOPTALIFESTYLE1403"
		event_target:male_mate = { set_artifact_flag = mated }
		event_target:female_mate = { set_artifact_flag = mated }
		
		# Trigger events for the holder of e_rebels to make sure these events fire consistently
		# even if something unexpected happens to the falconer
		hidden_tooltip = { e_rebels = { holder_scope = { character_event = { id = lifestyle.1404 days = 1095 random = 1460 } } } }
		ai_chance = { factor = 1 }
	}
	
	option = {
		# Disallow the mating
		name = "EVTOPTBLIFESTYLE1403"
		ai_chance = { factor = 0 }
	}
}

# Two mated peregrine falcons potentially breed
character_event = {
	id = lifestyle.1404
	hide_window = yes
	is_triggered_only = yes
	
	immediate = {
		if = {
			# Both falcons must be in the same place and fit to breed
			limit = {
				event_target:male_mate = {
					NOT = { has_artifact_flag = sick }
					NOT = { has_artifact_flag = injured }
					owner = {
						event_target:female_mate = {
							owner = { character = PREVPREV }
							NOT = { has_artifact_flag = sick }
							NOT = { has_artifact_flag = injured }
						}
					}
				}
			}
			event_target:male_mate = { owner = { character_event = { id = lifestyle.1405 } } }
		}
		# Continue repeating the breeding event as long as both falcons are alive
		if = {
			limit = {
				event_target:male_mate = { always = yes }
				event_target:female_mate = { always = yes }
			}
			character_event = { id = lifestyle.1404 days = 1095 random = 1460 }
		}
	}
}

					factor = 0.6
					event_target:training_raptor = { artifact_age >= 8 }
