# Assign a field to automatically-generated alumni
character_event = {
	id = school.1
	hide_window = yes
	is_triggered_only = yes
	
	trigger = {
		has_game_started = no
		FROM = { is_civilian_school_trigger = yes }
		age = 25
		NOT = { has_post_secondary_education_trigger = yes }
	}
	
	immediate = {
		if = {
			limit = { has_education_stewardship_trigger = yes }
			add_trait = economist
		}
		else_if = {
			limit = { has_education_diplomacy_trigger = yes }
			add_trait = lawyer
		}
		else = {
			random_list = {
				50 = { add_trait = physician }
				50 = { add_trait = scientist }
			}
		}
		if = {
			limit = { lifestyle_traits < 1 }
			random = {
				chance = 25
				add_trait = intellectual
			}
		}
	}
}

					limit = {

						trait = lawyer
					}

			}
		}
		else_if = {
			limit = { has_character_modifier = economics_major }

						society_rank >= 3

					}
					save_event_target_as = target_professor
				}
			}
		}

				factor = 0
				trait = disciplined
			}
		}
	}

		trigger = { NOT = { trait = stressed } }
		name = "EVTOPTBSCHOOL100"
		change_variable = { which = stress_var value = 20 }
		if = {
			limit = { trait = carefree }
			custom_tooltip = { text = "BECAUSECAREFREE" }
			change_variable = { which = stress_var value = 10 }
		}
		if = {
			limit = { trait = disciplined }
			custom_tooltip = { text = "BECAUSEDISCIPLINED" }
			change_variable = { which = stress_var value = -10 }
		}
		modify_stress_effect = yes
		ai_chance = {
			factor = 25
			modifier = {
				factor = 0
				trait = carefree
			}
		}
	}
	
	# Ask the professor for help
	option = {
		trigger = { event_target:target_professor = { always = yes } }

		if = {
			limit = { trait = introverted }
			custom_tooltip = { text = "BECAUSEINTROVERTED" }
			change_variable = { which = stress_var value = 10 }
		}
		if = {
