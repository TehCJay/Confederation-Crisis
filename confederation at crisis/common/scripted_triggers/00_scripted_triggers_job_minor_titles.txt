### Council Jobs ###

# Used with character interaction to swap council job. DO NOT REMOVE
# ROOT is councillor
# FROM is liege
change_council_job_potential_trigger = {
	liege = { character = FROM }

	trigger_if = {
		limit = {
			OR = {
				has_minor_title = title_senatorduke
				has_minor_title = title_senatorking
				has_minor_title = title_senatoremperor
				has_minor_title = title_senatorbigrealm1
				has_minor_title = title_senatorbigrealm2
				has_minor_title = title_senatorbigrealm3
				has_minor_title = title_senatorbigrealm4
				has_minor_title = title_senatorbigrealm5
			}
		}

		OR = {
			can_be_chancellor_trigger = yes
			can_be_marshal_trigger = yes
			can_be_treasurer_trigger = yes
			can_be_spymaster_trigger = yes
			can_be_spiritual_trigger = yes
		}
	}
	trigger_else = {
		OR = {
			has_job_title = job_chancellor
			has_job_title = job_marshal
			has_job_title = job_treasurer
			has_job_title = job_spymaster
			has_job_title = job_spiritual
		}

		liege = {
			OR = {
				job_chancellor = {
					NOT = { character = ROOT }
					can_swap_job_title = ROOT
				}
				job_marshal = {
					NOT = { character = ROOT }
					can_swap_job_title = ROOT
				}
				job_treasurer = {
					NOT = { character = ROOT }
					can_swap_job_title = ROOT
				}
				job_spymaster = {
					NOT = { character = ROOT }
					can_swap_job_title = ROOT
				}
				job_spiritual = {
					NOT = { character = ROOT }
					can_swap_job_title = ROOT
				}
			}
		}
	}
}

can_be_anything_trigger = {
	is_adult = yes
	independent = no
	prisoner = no
	is_incapable = no

	NOR = {
		trait = in_hiding
		has_character_modifier = resigned_in_anger

		liege = {
			has_opinion_modifier = {
				who = ROOT
				modifier = opinion_resigned_from_council
			}
		}
	}
}

gender_compatible_trigger = {
	trigger_if = {
		limit = { is_female = yes }

		liege = {
			OR = {
				has_religion_feature = religion_matriarchal
				has_religion_feature = religion_equal
				has_religion_feature = religion_feature_bon

				# Laws permitting
				primary_title = {
					trigger_if = {
						limit = {
							has_law = gender_status_0
						}

						ROOT = {
							OR = {
								is_close_relative = liege
								is_married = liege
								is_consort = liege
								#trait = nun
							}
						}
					}
					trigger_else_if = {
						limit = {
							has_law = gender_status_1
						}

						ROOT = { is_landed = yes }
						ROOT = {
							OR = {
								is_close_relative = liege
								is_married = liege
								is_consort = liege
								is_landed = yes
								#trait = nun
							}
						}
					}
					trigger_else = {
						OR = {
							has_law = gender_status_2
							has_law = gender_status_3
							has_law = gender_status_4
						}
					}
				}
			}
		}
	}
	trigger_else = {
		liege = {
			OR = {
				has_religion_feature = religion_patriarchal
				has_religion_feature = religion_equal

				# Laws permitting
				primary_title = {
					trigger_if = {
						limit = {
							has_law = gender_status_4
						}

						ROOT = {
							OR = {
								is_close_relative = liege
								is_married = liege
								is_consort = liege
								#trait = nun
							}
						}
					}
					trigger_else_if = {
						limit = {
							has_law = gender_status_3
						}

						ROOT = { is_landed = yes }
						ROOT = {
							OR = {
								is_close_relative = liege
								is_married = liege
								is_consort = liege
								is_landed = yes
								#trait = monk
							}
						}
					}
					trigger_else = {
						OR = {
							has_law = gender_status_2
							has_law = gender_status_1
							has_law = gender_status_0
						}
					}
				}
			}
		}
	}
}

can_be_spiritual_trigger = {
	can_be_anything_trigger = yes

	trigger_if = {
		limit = { NOT = { has_character_flag = special_spiritual } }


		trigger_if = {
			limit = {
				liege = {
					OR = {
						religion_group = religious_group
						religion_group = religious_terran_group
						religion_group = transhumanist_group
						religion_group = outlaw_group
						religion_group = political_terran_group
						religion_group = political_group
					}
				}
			}

			religion = liege

			OR = {
				trait = detached_priest
				trait = martial_cleric
				trait = scholarly_theologian
				trait = mastermind_theologian
			}

			trigger_if = {
				limit = { is_female = yes }

				OR = {
					religion_allows_female_temple_holders = yes
					has_religion_feature = religion_matriarchal
				}
			}
			trigger_else = {
				OR = {
					religion_allows_male_temple_holders = yes
					has_religion_feature = religion_patriarchal
				}
			}
		}
		trigger_else = {
			gender_compatible_trigger = yes
		}

		trigger_if = {
			limit = {
				is_theocracy = no
			}

			trigger_if = {
				limit = {
					liege = { is_nomadic = yes }
				}
				clan = yes
			}
			trigger_else_if = {
				limit = {
					liege = {
						has_autocephaly = yes
					}
				}
				liege = {
					NOT = { any_vassal = { is_theocracy = yes } }
				}
				is_ruler = no
			}
			trigger_else = {
				is_ruler = no
			}
		}
	}
}
