# ROOT = claimant, FROMFROM = attacker, THIS = target title
emf_cb_other_claimant_will_be_attacker_vassal = {
	FROMFROM = {
		primary_title = { higher_tier_than = PREVPREV } # Presser can bind target title
	}
	OR = {
		FROMFROM = {
			OR = {
				dynasty = ROOT # Presser is same dynasty as claimant
				any_demesne_title = { # Presser is de jure liege of target title, or...
					de_jure_vassal_or_below = PREVPREV
				}
			}
		}
		ROOT = { # Claimant is vassal ruler of presser and no less than one tier below target title
			is_ruler = yes
			is_liege_or_above = FROMFROM
			OR = {
				NOT = { lower_tier_than = PREV }
				PREV = { tier = COUNT }
				AND = {
					PREV = { tier = DUKE }
					tier = COUNT
				}
				AND = {
					PREV = { tier = KING }
					tier = DUKE
				}
			}
		}
	}
}

# ROOT = claimant, THIS = claimed title
emf_cb_can_press_claim = {
	ROOT = {
		OR = {
			is_ruler = no
			primary_title = {
				OR = {
					is_primary_type_title = no # Static Mercs, the Pope, Holy Orders, etc
					NOT = { lower_tier_than = PREVPREV }
				}
			}
		}
	}
	OR = {
		ROOT = { has_strong_claim = PREV }
		AND = {
			ROOT = { has_weak_claim = PREV }
			emf_cb_is_weak_claim_pressable = yes
		}
	}
}

# Shared `can_use` CB code for religious CBs
emf_cb_can_use_religious = {
	ROOT = {
		is_nomadic = no
		NOT = { same_realm = FROM }
		NOR = {
			mercenary = yes
			pacifist = yes
			religion = buddhist
		}
		emf_hostile_religion_to_FROM = yes
		OR = { # Allow reformed pagans [only]
			NOT = { religion_group = pagan_group }
			is_reformed_religion = yes
		}
		OR = { # Hindus can only wage holy war on non-Indian rulers
			NOT = { religion = hindu }
			FROM = { NOT = { religion_group = indian_group } }
			AND = {
				religion_group = FROM
				FROM = { has_character_modifier = known_satanist }
				NOT = { has_character_modifier = known_satanist }
			}
		}
		top_liege = {
			religion_group = ROOT
			OR = {
				NOT = { religion = FROM }
				AND = {
					religion_group = FROM
					FROM = { has_character_modifier = known_satanist }
					NOT = { has_character_modifier = known_satanist }
					NOT = { ROOT = { has_character_modifier = known_satanist } }
				}
			}
			OR = {
				religion = ROOT
				AND = {
					is_heretic = no
					ROOT = { is_heretic = no }
				}
			}
			OR = {
				NOT = { religion_group = pagan_group }
				religion = ROOT
			}
		}
	}
}

			}
		}
	}
}

# by event and thus the succ_adopt_* effects should skip these characters in its blanket opinion effects. Note that we

	}
}

	}
}

	}
	# Female claimants must be ambitious and can only target non-agnatic titles
	OR = {
		ROOT = { has_character_flag = event_adventurer }
		AND = {
			ROOT = { is_female = no }
			NOT = { has_law = enatic_succession }
		}
		AND = {
			ROOT = { is_female = yes }
			ROOT = { trait = ambitious }
			NOT = { has_law = agnatic_succession }
		}
	}
	# Title holder must qualify
	holder_scope = {
		independent = yes
		demesne_size = 1
		# Not held by theocracy unless not of the claimant's religion
		OR = {
			is_theocracy = no
			NOT = { religion = ROOT }
		}
		# Cannot target nomads for their clan title or primary title
		NAND = {
			government = nomadic_government
			OR = {
				clan_title = { title = PREVPREV }
				primary_title = { title = PREVPREV }
			}
		}
		# Cannot target merchant republics unless their capital is not de jure
		OR = {
			is_merchant_republic = no
			NOT = { capital_holding = { de_jure_liege_or_above = PREVPREV } }
		}
		# Cannot target close relatives unless they are hated or the claimant is a bastard
		OR = {
			NOT = { reverse_opinion = { who = ROOT value = -25 } }
			NOT = { is_close_relative = ROOT }
			ROOT = { trait = bastard }
		}
		NOR = {
			in_revolt = yes
			any_liege = { in_revolt = yes }
			any_spouse = { character = ROOT }
			reverse_opinion = { who = ROOT value = 25 }

		}
		AND = {
			event_target:faction_leader = { leads_faction = faction_tradition }

		}
	}
}
