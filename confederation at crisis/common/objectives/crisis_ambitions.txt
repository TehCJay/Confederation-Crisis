obj_expand_realm = { # Expand Realm via invasion
	type = character

	aggression = 4.0

	potential = {
		independent = yes
		NOT = { has_character_flag = obj_expand_realm }
		has_game_rule = { name = ambition_invasion value = normal }
		OR = {
			ai = yes
			multiplayer = no
			NOT = {
				has_game_rule = {
					name = multiplayer_invasions
					value = off
				}
			}
		}
	}
	allow = {
		#ai = no
		prisoner = no
		is_adult = yes

		demesne_size = 1
		num_of_count_titles = 1
		custom_tooltip = {
			text = TOOLTIPINDEPENDENTCONDITION
			hidden_tooltip = { independent = yes }
		}
		trait = ambitious
		is_incapable = no
		NOT = { trait = content }
		can_press_claims_trigger = yes
		is_nomadic = no
	}
	chance = {
		factor = 50

		modifier = {
			factor = 0
			NOT = { prestige = 1000 }
		}
		modifier = {
			factor = 0.1
			tier = COUNT
		}
		modifier = {
			factor = 0.2
			tier = DUKE
		}
		modifier = {
			factor = 0.5
			trait = craven
		}
		modifier = {
			factor = 2
			trait = brave
		}
		modifier = {
			factor = 0.1
			trait = humble
		}
		modifier = {
			factor = 10.0
			trait = ambitious
		}
		modifier = {
			factor = 2.0
			trait = cold
		}
		modifier = {
			factor = 2.0
			trait = proud
		}
	}
	success = {
		custom_tooltip = {
			text = obj_expand_realm_SUCCESS
			hidden_tooltip = { has_character_flag = obj_expand_realm_success }
		}
	}
	abort = {
		OR = {
			AND = {
				has_character_flag = abort_obj
				NOT = { has_character_flag = called_grand_council }
				NOT = { has_character_flag = may_call_ambition_invasion }
			}
			trait = incapable
			is_ruler = no
			independent = no
			AND = {
				war = no
				had_character_flag = { flag = may_call_ambition_invasion days = 1825 }
			}
			NOT = { can_press_claims_trigger = yes }
		}
	}
	abort_effect = {
		clr_character_flag = abort_obj
		clr_character_flag = obj_check
		clr_character_flag = called_grand_council
		clr_character_flag = may_call_ambition_invasion
	}
	effect = {
		prestige = 200
		if = {
			limit = { has_nickname = no }
			give_nickname = nick_the_conqueror
		}
		clr_character_flag = obj_expand_realm_success
		clr_character_flag = called_grand_council
		clr_character_flag = may_call_ambition_invasion
		clr_character_flag = abort_obj
	}
}
