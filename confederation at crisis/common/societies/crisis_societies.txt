# Civilian Schools
university_of_the_confederation = {
	primary_attribute = learning
	opinion_to_other_members = +10
	opinion_per_rank_above = +5
	sound = hermetic_society_interface
	society_rank_up_decision = civilian_school_graduation_decision
	active = { NOT = { has_global_flag = flag_no_university_of_the_confederation } }
	
	show_society = { c_sol = { holder_scope = { is_within_diplo_range = ROOT } } }
	can_join_society = {
		age = 18
		NOT = { religion_group = outlaw_group }
		NOT = { has_character_flag = flag_expelled_university_of_the_confederation }
		OR = {
			has_education_diplomacy_trigger = yes
			has_education_stewardship_trigger = yes
			has_education_learning_trigger = yes
		}
	}
	potential = {
		NOT = { has_character_flag = flag_expelled_university_of_the_confederation }
		NOT = { c_sol = { holder_scope = { war_with = ROOT } } }
	}
	
	# Student
	society_rank = {
		level = 1
		limit = 200
		startup_limit = 50
		modifier = {
			monthly_character_wealth = -0.2
		}
	}
	
	# Associate
	society_rank = {
		level = 2
		limit = 500
		startup_limit = 50
		modifier = {
			monthly_character_wealth = -0.1
			monthly_character_prestige = 0.1
		}
	}
	
	# Professor
	society_rank = {
		level = 3
		limit = 20
		startup_limit = 10
		modifier = {
			monthly_character_wealth = 0.1
			monthly_character_prestige = 0.1
		}
	}
	
	# Rector
	society_rank = {
		level = 4
		limit = 1
		startup_limit = 1
		modifier = {
			monthly_character_wealth = 0.2
			monthly_character_prestige = 0.25
		}
	}
	
	monthly_currency_gain = {
		name = currency_name_university_of_the_confederation
		per_attribute = {
			name = learning
			value = 1
		}
		has_trait = {
			trait = open_minded
			value = 5
		}
		has_trait = {
			trait = introverted
			value = 5
		}
		has_trait = {
			trait = disciplined
			value = 5
		}
		has_trait = {
			trait = honest
			value = 5
		}
		has_trait = {
			trait = intellectual
			value = 10
		}
		triggered_gain = {
			trigger = {
				society_rank >= 2
				society = {
					any_society_member = {
						is_society_grandmaster = yes
						opinion = { who = ROOT value = 50 }
					}
				}
			}
			desc = "GRANDMASTERSFAVOR"
			value = 10
		}
	}

	member_score_per_attribute = 3
	member_score_per_rank = 100

	member_score = {
		value = 10
		
		modifier = {
			factor = 10
			OR = {
				trait = lawyer
				trait = economist
				trait = physician
				trait = scientist
			}
		}
		modifier = {
			factor = 10
			trait = intellectual
		}
		
		modifier = {
			factor = 2
			crownlaw_title = {
				OR = {
					is_landed_title = e_terran_confederation
					is_landed_title = k_terran_commonwealth
				}
			}
		}
		modifier = {
			factor = 4
			culture_group = terran_group
		}
		modifier = {
			factor = 2
			trait = disciplined
		}
		modifier = {
			factor = 2
			trait = open_minded
		}
		modifier = {
			factor = 2
			trait = confident
		}
	}

	startup_populate = {
	    trigger = {
			ai = yes
			historical = no # Historical members should be added manually
	    	controls_religion = no
			age = 18
			is_in_society = no
			is_dumb_trigger = no
			NOT = { religion_group = outlaw_group }
			OR = {
				c_huangjing = { same_realm = ROOT }
				culture_group = terran_group
			}
			OR = {
				has_education_diplomacy_trigger = yes
				has_education_stewardship_trigger = yes
				has_education_learning_trigger = yes
			}
			OR = {
				diplomacy = 8
				stewardship = 8
				learning = 12
				NOT = { age = 25 }
			}
			learning = 8
		}
	}
}

	opinion_to_other_members = +10
	opinion_per_rank_above = +5

	society_rank_up_decision = civilian_school_graduation_decision
	active = { NOT = { has_global_flag = flag_no_university_of_armstrong } }
	
	show_society = { c_armstrong = { holder_scope = { is_within_diplo_range = ROOT } } }
	can_join_society = {
		age = 18
		NOT = { religion_group = outlaw_group }
		NOT = { has_character_flag = flag_expelled_university_of_armstrong }
		OR = {
			has_education_diplomacy_trigger = yes
			has_education_stewardship_trigger = yes
			has_education_learning_trigger = yes
		}
	}
	potential = {
		NOT = { has_character_flag = flag_expelled_university_of_armstrong }
		NOT = { c_armstrong = { holder_scope = { war_with = ROOT } } }
	}
	
	# Student
	society_rank = {
		level = 1
		limit = 200
		startup_limit = 50
		modifier = {
			monthly_character_wealth = -0.2
		}
	}
	
	# Associate
	society_rank = {
		level = 2
		limit = 500
		startup_limit = 50
		modifier = {
			monthly_character_wealth = -0.1
			monthly_character_prestige = 0.1
		}
	}
	
	# Professor
	society_rank = {
		level = 3
		limit = 20
		startup_limit = 10
		modifier = {
			monthly_character_wealth = 0.1
			monthly_character_prestige = 0.1
		}
	}
	
	# Rector
	society_rank = {
		level = 4
		limit = 1
		startup_limit = 1
		modifier = {
			monthly_character_wealth = 0.2
			monthly_character_prestige = 0.25
		}
	}
	
	monthly_currency_gain = {
		name = currency_name_university_of_armstrong
		per_attribute = {
			name = learning
			value = 1
		}
		has_trait = {
			trait = open_minded
			value = 5
		}
		has_trait = {
			trait = introverted
			value = 5
		}
		has_trait = {
			trait = disciplined
			value = 5
		}
		has_trait = {
			trait = honest
			value = 5
		}
		has_trait = {
			trait = intellectual
			value = 10
		}
		triggered_gain = {
			trigger = {
				society_rank >= 2
				society = {
					any_society_member = {
						is_society_grandmaster = yes
						opinion = { who = ROOT value = 50 }
					}
				}
			}
			desc = "GRANDMASTERSFAVOR"
			value = 10
		}
	}

	member_score_per_attribute = 3
	member_score_per_rank = 100

	member_score = {
		value = 10
		
		modifier = {
			factor = 10
			OR = {
				trait = lawyer
				trait = economist
				trait = physician
				trait = scientist
			}
		}
		modifier = {
			factor = 10
			trait = intellectual
		}
		
		modifier = {
			factor = 2
			crownlaw_title = {
				OR = {
					is_landed_title = e_terran_confederation
					is_landed_title = e_frontier
					is_landed_title = k_frontier
				}
			}
		}
		modifier = {
			factor = 4
			culture_group = frontiersman_group
		}
		modifier = {
			factor = 2
			trait = disciplined
		}
		modifier = {
			factor = 2
			trait = open_minded
		}
		modifier = {
			factor = 2
			trait = confident
		}
	}

	startup_populate = {
	    trigger = {
			ai = yes
			historical = no # Historical members should be added manually
	    	controls_religion = no
			age = 18
			is_in_society = no
			is_dumb_trigger = no
			NOT = { religion_group = outlaw_group }
			OR = {
				c_armstrong = { same_realm = ROOT }
				culture_group = frontiersman_group
			}
			OR = {
				has_education_diplomacy_trigger = yes
				has_education_stewardship_trigger = yes
				has_education_learning_trigger = yes
			}
			OR = {
				diplomacy = 10
				stewardship = 10
				learning = 15
				NOT = { age = 25 }
			}
			learning = 10
		}
	}
}
